20  static cv::Mat Thinning(cv::Mat image) {
21  //https://github.com/bsdnoobz/zhang-suen-thinning
22  CV_Assert(image.channels() == 1);
23  CV_Assert(image.depth() != sizeof(uchar));
24  CV_Assert(image.rows > 3 && image.cols > 3);
25  cv::Mat ret = image.clone();
26  ret /= 255;
27  cv::Mat prev_img = cv::Mat::zeros(ret.size(), CV_8UC1);
28  cv::Mat marker = cv::Mat::zeros(ret.size(), CV_8UC1);
29  cv::Mat diff;
30  do {
31  for (int iter = 0; iter < 2; iter++) {
32  marker.setTo(0);
33  int nRows = ret.rows;
34  int nCols = ret.cols;
35  if (ret.isContinuous() == true) {
36  nCols *= nRows;
37  nRows = 1;
38  }
39  int x, y;
40  uchar *pAbove;
41  uchar *pCurr;
42  uchar *pBelow;
43  uchar *nw, *no, *ne; // north (pAbove)
44  uchar *we, *me, *ea;
45  uchar *sw, *so, *se; // south (pBelow)
46  uchar *pDst;
47  pAbove = nullptr;
48  pCurr = ret.ptr<uchar>(0);
49  pBelow = ret.ptr<uchar>(1);
50  for (y = 1; y < ret.rows - 1; ++y) {
51  pAbove = pCurr;
52  pCurr = pBelow;
53  pBelow = ret.ptr<uchar>(y + 1);
54  pDst = marker.ptr<uchar>(y);
55  no = &(pAbove[0]);
56  ne = &(pAbove[1]);
57  me = &(pCurr[0]);
58  ea = &(pCurr[1]);
59  so = &(pBelow[0]);
60  se = &(pBelow[1]);
61  for (x = 1; x < ret.cols - 1; ++x) {
62  nw = no;
63  no = ne;
64  ne = &(pAbove[x + 1]);
65  we = me;
66  me = ea;
67  ea = &(pCurr[x + 1]);
68  sw = so;
69  so = se;
70  se = &(pBelow[x + 1]);
71  int A = (*no == 0 && *ne == 1) + (*ne == 0 && *ea == 1) +
72  (*ea == 0 && *se == 1) + (*se == 0 && *so == 1) +
73  (*so == 0 && *sw == 1) + (*sw == 0 && *we == 1) +
74  (*we == 0 && *nw == 1) + (*nw == 0 && *no == 1);
75  int B = *no + *ne + *ea + *se + *so + *sw + *we + *nw;
76  int m1 = iter == 0 ? (*no * *ea * *so) : (*no * *ea * *we);
77  int m2 = iter == 0 ? (*ea * *so * *we) : (*no * *so * *we);
78  if (A == 1 && (B >= 2 && B <= 6) && m1 == 0 && m2 == 0)
79  pDst[x] = 1;
80  }
81  }
82  ret &= ~marker;
83  }
84  cv::absdiff(ret, prev_img, diff);
85  prev_img = ret.clone();
86  } while (cv::countNonZero(diff) > 0);
87  ret *= 255;
88  for (int y = 1; y < ret.rows - 1; y++) {
89  for (int x = 1; x < ret.cols - 1; x++) {
90  bool up = ret.at<uchar>(y - 1, x) == 255;
91  bool down = ret.at<uchar>(y + 1, x) == 255;
92  bool left = ret.at<uchar>(y, x - 1) == 255;
93  bool right = ret.at<uchar>(y, x + 1) == 255;
94  int c = (up&right) + (up&left) + (down&left) + (down&right);
95  if (c==1) {
96  ret.at<uchar>(y, x) = 0;
97  }
98  }
99  }
100  return ret;
101  }
102  static cv::Mat AutoCanny(cv::Mat image) {
103  cv::Mat gray, edge, dummy;
104  if (image.channels() == 3) {
105  cv::cvtColor(image, gray, CV_BGR2GRAY);
106  }
107  else {
108  gray = image;
109  }
110  double high_th = cv::threshold(gray, dummy, 0, 255, cv::THRESH_BINARY | cv::THRESH_OTSU);
111  double low_th = high_th / 2.;
112  cv::Canny(gray, edge, low_th, high_th);
113  return edge;

149  static cv::Mat Zoom(cv::Mat image, double ratio) {
150  if (ratio == 1.0F) {
151  return image.clone();
152  } else if (ratio < 1.0F) {

154  cv::Mat zoomed = cv::Mat::zeros(image.size(), CV_8UC3);
155  cv::Mat small_img;

157  int left_padding = (image.cols - small_img.cols) / 2;

159  small_img.copyTo(zoomed(cv::Rect(left_padding, top_padding, small_img.cols, small_img.rows)));
160  return zoomed;
161  } else { //ratio>1.0F
162  cv::Size zsize(static_cast<int>(ratio*image.cols), static_cast<int>(ratio*image.rows));
163  cv::Mat zoomed = cv::Mat::zeros(image.size(), CV_8UC3);
164  cv::Mat large_img;
165  cv::resize(image, large_img, zsize);
166  int left_margin = (large_img.cols - image.cols) / 2;
167  int top_margin = (large_img.rows - image.rows) / 2;
168  zoomed = large_img(cv::Rect(left_margin, top_margin, large_img.cols - left_margin * 2, large_img.rows - top_margin * 2));
169  return zoomed(cv::Rect(0, 0, image.cols, image.rows));

172  static cv::Mat ImageRotateOuter(const cv::Mat image, double degree, cv::Point* out = nullptr) {
173  cv::Point2d base(image.cols / 2.0, image.rows / 2.0);
174  cv::Mat rot = cv::getRotationMatrix2D(base, degree, 1.0);
175  cv::Rect bbox = cv::RotatedRect(base, image.size(),(float) degree).boundingRect();
176  rot.at<double>(0, 2) += bbox.width / 2.0 - base.x;
177  rot.at<double>(1, 2) += bbox.height / 2.0 - base.y;
178  if (out != nullptr) {
179  out->x = static_cast<int>(bbox.width / 2.0 - base.x);
180  out->y = static_cast<int>( bbox.height / 2.0 - base.y);
181  }
182  cv::Mat dst;
183  cv::warpAffine(image, dst, rot, bbox.size());
184  return std::move(dst);
185  }
