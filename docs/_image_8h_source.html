42  static cv::Mat Thinning(cv::Mat image) {
43  //https://github.com/bsdnoobz/zhang-suen-thinning
44  if (image.channels() != 1 || image.depth() != 0) {
45  ISPRING_VERIFY("This is not binary image");
46  }
47  if (image.rows <= 3 || image.cols <= 3) {
48  ISPRING_VERIFY("Image size is too small");
49  }
50  cv::Mat ret = image.clone();
51  ret /= 255;
52  cv::Mat prev_img = cv::Mat::zeros(ret.size(), CV_8UC1);
53  cv::Mat marker = cv::Mat::zeros(ret.size(), CV_8UC1);
54  cv::Mat diff;
55  do {
56  for (int iter = 0; iter < 2; iter++) {
57  marker.setTo(0);
58  int nRows = ret.rows;
59  int nCols = ret.cols;
60  if (ret.isContinuous() == true) {
61  nCols *= nRows;
62  nRows = 1;
63  }
64  int x, y;
65  uchar *pAbove;
66  uchar *pCurr;
67  uchar *pBelow;
68  uchar *nw, *no, *ne; // north (pAbove)
69  uchar *we, *me, *ea;
70  uchar *sw, *so, *se; // south (pBelow)
71  uchar *pDst;
72  pAbove = nullptr;
73  pCurr = ret.ptr<uchar>(0);
74  pBelow = ret.ptr<uchar>(1);
75  for (y = 1; y < ret.rows - 1; ++y) {
76  pAbove = pCurr;
77  pCurr = pBelow;
78  pBelow = ret.ptr<uchar>(y + 1);
79  pDst = marker.ptr<uchar>(y);
80  no = &(pAbove[0]);
81  ne = &(pAbove[1]);
82  me = &(pCurr[0]);
83  ea = &(pCurr[1]);
84  so = &(pBelow[0]);
85  se = &(pBelow[1]);
86  for (x = 1; x < ret.cols - 1; ++x) {
87  nw = no;
88  no = ne;
89  ne = &(pAbove[x + 1]);
90  we = me;
91  me = ea;
92  ea = &(pCurr[x + 1]);
93  sw = so;
94  so = se;
95  se = &(pBelow[x + 1]);
96  int A = (*no == 0 && *ne == 1) + (*ne == 0 && *ea == 1) +
97  (*ea == 0 && *se == 1) + (*se == 0 && *so == 1) +
98  (*so == 0 && *sw == 1) + (*sw == 0 && *we == 1) +
99  (*we == 0 && *nw == 1) + (*nw == 0 && *no == 1);
100  int B = *no + *ne + *ea + *se + *so + *sw + *we + *nw;
101  int m1 = iter == 0 ? (*no * *ea * *so) : (*no * *ea * *we);
102  int m2 = iter == 0 ? (*ea * *so * *we) : (*no * *so * *we);
103  if (A == 1 && (B >= 2 && B <= 6) && m1 == 0 && m2 == 0)
104  pDst[x] = 1;
105  }
106  }
107  ret &= ~marker;
108  }
109  cv::absdiff(ret, prev_img, diff);
110  prev_img = ret.clone();
111  } while (cv::countNonZero(diff) > 0);
112  ret *= 255;
113  for (int y = 1; y < ret.rows - 1; y++) {
114  for (int x = 1; x < ret.cols - 1; x++) {
115  bool up = ret.at<uchar>(y - 1, x) == 255;
116  bool down = ret.at<uchar>(y + 1, x) == 255;
117  bool left = ret.at<uchar>(y, x - 1) == 255;
118  bool right = ret.at<uchar>(y, x + 1) == 255;
119  int c = (up&right) + (up&left) + (down&left) + (down&right);
120  if (c==1) {
121  ret.at<uchar>(y, x) = 0;
122  }
123  }
124  }
125  return ret;
126  }
