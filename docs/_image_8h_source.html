38  static cv::Mat Thinning(cv::Mat image) {
39  //https://github.com/bsdnoobz/zhang-suen-thinning
40  if (image.channels() != 1 || image.depth() != 0) {
41  ISPRING_VERIFY("This is not binary image");
42  }
43  if (image.rows <= 3 || image.cols <= 3) {
44  ISPRING_VERIFY("Image size is too small");
45  }
46  cv::Mat ret = image.clone();
47  ret /= 255;
48  cv::Mat prev_img = cv::Mat::zeros(ret.size(), CV_8UC1);
49  cv::Mat marker = cv::Mat::zeros(ret.size(), CV_8UC1);
50  cv::Mat diff;
51  do {
52  for (int iter = 0; iter < 2; iter++) {
53  marker.setTo(0);
54  int nRows = ret.rows;
55  int nCols = ret.cols;
56  if (ret.isContinuous() == true) {
57  nCols *= nRows;
58  nRows = 1;
59  }
60  int x, y;
61  uchar *pAbove;
62  uchar *pCurr;
63  uchar *pBelow;
64  uchar *nw, *no, *ne; // north (pAbove)
65  uchar *we, *me, *ea;
66  uchar *sw, *so, *se; // south (pBelow)
67  uchar *pDst;
68  pAbove = nullptr;
69  pCurr = ret.ptr<uchar>(0);
70  pBelow = ret.ptr<uchar>(1);
71  for (y = 1; y < ret.rows - 1; ++y) {
72  pAbove = pCurr;
73  pCurr = pBelow;
74  pBelow = ret.ptr<uchar>(y + 1);
75  pDst = marker.ptr<uchar>(y);
76  no = &(pAbove[0]);
77  ne = &(pAbove[1]);
78  me = &(pCurr[0]);
79  ea = &(pCurr[1]);
80  so = &(pBelow[0]);
81  se = &(pBelow[1]);
82  for (x = 1; x < ret.cols - 1; ++x) {
83  nw = no;
84  no = ne;
85  ne = &(pAbove[x + 1]);
86  we = me;
87  me = ea;
88  ea = &(pCurr[x + 1]);
89  sw = so;
90  so = se;
91  se = &(pBelow[x + 1]);
92  int A = (*no == 0 && *ne == 1) + (*ne == 0 && *ea == 1) +
93  (*ea == 0 && *se == 1) + (*se == 0 && *so == 1) +
94  (*so == 0 && *sw == 1) + (*sw == 0 && *we == 1) +
95  (*we == 0 && *nw == 1) + (*nw == 0 && *no == 1);
96  int B = *no + *ne + *ea + *se + *so + *sw + *we + *nw;
97  int m1 = iter == 0 ? (*no * *ea * *so) : (*no * *ea * *we);
98  int m2 = iter == 0 ? (*ea * *so * *we) : (*no * *so * *we);
99  if (A == 1 && (B >= 2 && B <= 6) && m1 == 0 && m2 == 0)
100  pDst[x] = 1;
101  }
102  }
103  ret &= ~marker;
104  }
105  cv::absdiff(ret, prev_img, diff);
106  prev_img = ret.clone();
107  } while (cv::countNonZero(diff) > 0);
108  ret *= 255;
109  for (int y = 1; y < ret.rows - 1; y++) {
110  for (int x = 1; x < ret.cols - 1; x++) {
111  bool up = ret.at<uchar>(y - 1, x) == 255;
112  bool down = ret.at<uchar>(y + 1, x) == 255;
113  bool left = ret.at<uchar>(y, x - 1) == 255;
114  bool right = ret.at<uchar>(y, x + 1) == 255;
115  int c = (up&right) + (up&left) + (down&left) + (down&right);
116  if (c==1) {
117  ret.at<uchar>(y, x) = 0;
118  }
119  }
120  }
121  return ret;
122  }
