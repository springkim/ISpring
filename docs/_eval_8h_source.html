81  static std::pair<float,float> GetRecallPrecisionSingleClass(std::vector<BoxSE> ground_truth,std::vector<BoxSE> predict,float threshold1=0.5F) {
82  std::pair<float, float> ret;
83  std::vector<int> correct;
84  correct.assign(predict.size(), -1);
85  for (size_t i = 0; i < predict.size(); ++i) {
86  for (size_t j = 0; j < ground_truth.size(); ++j) {
87  if (IOU(predict[i], ground_truth[j]) > threshold1) {
88  if (correct[i] == -1 || ground_truth[correct[i]].m_score < ground_truth[j].m_score) {
89  correct[i] = j;
90  }
91  }
92  }
93  }
94  std::vector<bool> bloom;
95  bloom.assign(ground_truth.size(), false);
96  for (auto&e : correct) {
97  if (e != -1) {
98  bloom[e] = true;
99  }
100  }
101  int TP = std::count(bloom.begin(), bloom.end(), true);
102  int FN = std::count(bloom.begin(), bloom.end(), false);
103  int FP = std::count(correct.begin(), correct.end(), -1);
104  ret.first = (float)TP / (TP + FN);
105  ret.second = (float)TP / (TP + FP);
106  return ret;
107  }
